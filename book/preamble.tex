% xltabular = tabularx + longtable: page-breakable two-column Q/A layout,
% with a colored vertical rule between columns and a thin colored horizontal
% rule between Q/A pairs.
\NewEnviron{qa}{%
  \setlength{\extrarowheight}{2pt}%
  \begin{xltabular}{\linewidth}{@{}>{\raggedright\arraybackslash}X@{\hspace{0.6em}}!{\color{DialogBorder!50}\vrule}@{\hspace{0.6em}}>{\raggedright\arraybackslash}X@{}}
    \BODY
  \end{xltabular}%
}
% \qa is taken by an internal command; use \qaitem instead.
\newcommand{\qaitem}[2]{%
  #1 & #2 \\[0.3em]%
  \noalign{{\color{DialogBorder!50}\hrule height 0.3pt}\vspace{0.6em}}%
}
