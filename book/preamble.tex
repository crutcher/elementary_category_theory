% ── Cycle-step label commands ────────────────────────────────────────────────
% Each label is a colored chip placed before a subsection heading.
% They signal which phase of the learning cycle the subsection occupies.
% Labels are drawn from a small vocabulary and reused across chapters.

\newcommand{\CycleLabel}[2]{%
  \vspace{0.5em}%
  \noindent\colorbox{#1}{\color{white}\footnotesize\bfseries\strut\ #2\ }%
  \vspace{0.25em}\par%
}

\newcommand{\InformalLabel}   {\CycleLabel{InformalColor}   {INFORMALLY}}
\newcommand{\ExpandingLabel}  {\CycleLabel{ExpandingColor}  {EXPANDING}}
\newcommand{\FormalizingLabel}{\CycleLabel{FormalizingColor}{FORMALIZING}}
\newcommand{\ReorganizingLabel}{\CycleLabel{ReorganizingColor}{REORGANIZING}}
\newcommand{\SimplifyingLabel}{\CycleLabel{SimplifyingColor}{SIMPLIFYING}}
\newcommand{\FormallyLabel}   {\CycleLabel{FormallyColor}   {FORMALLY}}
