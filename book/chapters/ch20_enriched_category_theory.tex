\chapter{Enriched Category Theory}

\begin{overviewbox}
Chapter~12 introduced $\mathcal{V}$-enriched categories, where hom-sets are
replaced by objects of a monoidal category $\mathcal{V}$. This chapter develops
the deep theory: the enriched Yoneda lemma, enriched adjunctions, weighted
limits and colimits, base change, and self-enrichment of closed categories.

The enriched Yoneda lemma is the heart of the chapter. It states that the
``set of natural transformations from a representable $\mathcal{V}$-functor to
$F$'' is (up to the right notion of $\mathcal{V}$-natural transformation) just
$F(a)$ in $\mathcal{V}$ --- the same statement as ordinary Yoneda, but now
the conclusion lands in $\mathcal{V}$, not $\mathbf{Set}$.
\end{overviewbox}

\section{The Enriched Yoneda Lemma}
\label{sec:enriched-yoneda}

\begin{theorem}[$\mathcal{V}$-Yoneda Lemma]
\label{thm:enriched-yoneda}
Let $\mathcal{C}$ be a $\mathcal{V}$-enriched category and $F : \mathcal{C}
\to \mathcal{V}$ a $\mathcal{V}$-functor. For each $a \in \mathcal{C}$:
\begin{equation*}
  [\mathcal{C}, \mathcal{V}]_\mathcal{V}(\mathcal{C}(a,-),\; F)
  \;=\;
  \int_{c} \mathcal{V}(\mathcal{C}(a,c), F(c))
  \;\cong\;
  F(a),
\end{equation*}
$\mathcal{V}$-naturally in $a$.
\end{theorem}

\begin{prooflog}
\proofstep{Construct $\phi : \int_c \mathcal{V}(\mathcal{C}(a,c), F(c)) \to F(a)$
by evaluating at $j_a : I \to \mathcal{C}(a,a)$}{The enriched identity morphism}
\proofstep{Construct inverse $\psi : F(a) \to \int_c \mathcal{V}(\mathcal{C}(a,c), F(c))$
by $\psi(x)_c = F_{a,c}(x \otimes \mathrm{id})$}{The $\mathcal{V}$-functor action}
\proofstep{$\phi \circ \psi = \mathrm{id}$ by the unit axiom for $\mathcal{V}$-functors}{}
\proofstep{$\psi \circ \phi = \mathrm{id}$ by the end condition (naturality of $\theta$)}{}
\end{prooflog}

\begin{corollary}[Enriched Yoneda Embedding]
The enriched Yoneda embedding $Y : \mathcal{C} \to [\mathcal{C}^{\mathrm{op}},
\mathcal{V}]_\mathcal{V}$ is fully faithful as a $\mathcal{V}$-functor.
\end{corollary}

\section{$\mathcal{V}$-Adjunctions}
\label{sec:enriched-adjunctions}

\begin{definition}
A \term{$\mathcal{V}$-adjunction} $F \dashv_\mathcal{V} G$ consists of a
$\mathcal{V}$-natural isomorphism
\begin{equation*}
  \varphi_{A,B} : \mathcal{D}(FA, B) \;\xrightarrow{\;\cong\;}\; \mathcal{C}(A, GB)
\end{equation*}
for all $A \in \mathcal{C}$, $B \in \mathcal{D}$, $\mathcal{V}$-natural in both.
Every $\mathcal{V}$-adjunction has an underlying ordinary adjunction (apply $\mathcal{V}(I,-)$).
\end{definition}

\section{Weighted Limits and Colimits}
\label{sec:weighted-limits}

\begin{definition}
The \term{$W$-weighted limit} $\{W, F\}$ of $F : \mathcal{J} \to \mathcal{C}$
weighted by $W : \mathcal{J}^{\mathrm{op}} \to \mathcal{V}$ satisfies:
\begin{equation*}
  \mathcal{C}(c, \{W, F\})
  \;\cong\;
  [\mathcal{J}^{\mathrm{op}}, \mathcal{V}]_\mathcal{V}(W,\; \mathcal{C}(c, F(-))).
\end{equation*}
Ordinary limits are $\Delta I$-weighted. In tensored/cotensored $\mathcal{C}$:
$\{W,F\} \cong \lim_j [W(j), F(j)]$.
\end{definition}

\section{Self-Enrichment of Closed Symmetric Monoidal Categories}
\label{sec:self-enrichment}

A \term{closed symmetric monoidal category} $(\mathcal{V}, \otimes, I, [-,-])$
where $[A, B]$ is the internal hom enriches over itself: set
$\mathcal{V}(A,B) = [A,B]$ with composition $[B,C]\otimes[A,B] \to [A,C]$
via evaluation.

\begin{example}{$\mathbf{Ab}$-enrichment}
$(\mathbf{Ab}, \otimes_\mathbb{Z}, \mathbb{Z})$ is closed with $[A,B] = \mathrm{Hom}(A,B)$.
\end{example}

\begin{example}{$\mathbf{sSet}$-enrichment}
$\mathbf{sSet}$ is closed monoidal with mapping space $\mathbf{Map}(X,Y)_n =
\mathbf{sSet}(X \times \Delta^n, Y)$.
\end{example}

\section*{Exercises}

\begin{exercisesbox}
\begin{qa}
\qaitem{State the enriched Yoneda lemma.}{$[\mathcal{C},\mathcal{V}]_\mathcal{V}(\mathcal{C}(a,-), F) \cong F(a)$ in $\mathcal{V}$, $\mathcal{V}$-naturally in $a$.}
\qaitem{What is a $\mathcal{V}$-adjunction?}{A $\mathcal{V}$-natural isomorphism $\mathcal{D}(FA,B) \cong \mathcal{C}(A,GB)$ in $\mathcal{V}$.}
\qaitem{What is a $W$-weighted limit?}{An object $\{W,F\} \in \mathcal{C}$ with $\mathcal{C}(c,\{W,F\}) \cong [\mathcal{J}^{\mathrm{op}},\mathcal{V}]_\mathcal{V}(W, \mathcal{C}(c,F-))$ $\mathcal{V}$-naturally in $c$.}
\qaitem{What ordinary limit corresponds to $W = \Delta I$?}{The ordinary conical limit $\lim F$.}
\qaitem{What is a closed symmetric monoidal category?}{One where $(-\otimes A)$ has a right adjoint $[A,-]$ and the tensor is symmetric.}
\qaitem{What is self-enrichment?}{Using $[A,B]$ as the hom-object for the $\mathcal{V}$-category structure on $\mathcal{V}$ itself.}
\qaitem{What is the change of base for enriched categories?}{A lax monoidal $\Phi : \mathcal{V} \to \mathcal{W}$ induces $(-)_\Phi$ sending $\mathcal{C}$ to $\mathcal{C}_\Phi$ with hom-objects $\Phi(\mathcal{C}(a,b))$.}
\end{qa}
\end{exercisesbox}

\section*{Chapter Summary}
\addcontentsline{toc}{section}{Chapter Summary}

\begin{summarybox}
\textbf{Enriched Yoneda.} $[\mathcal{C},\mathcal{V}]_\mathcal{V}(\mathcal{C}(a,-), F)
\cong F(a)$ in $\mathcal{V}$.

\textbf{$\mathcal{V}$-adjunctions.} Hom-set bijection replaced by
$\mathcal{V}$-natural isomorphism $\mathcal{D}(FA,B) \cong \mathcal{C}(A,GB)$.

\textbf{Weighted limits.} $\mathcal{C}(c, \{W,F\}) \cong
[\mathcal{J}^{\mathrm{op}},\mathcal{V}]_\mathcal{V}(W, \mathcal{C}(c, F-))$.

\textbf{Self-enrichment.} A closed symmetric monoidal $\mathcal{V}$ enriches
over itself using $[A,B]$ as the hom-object.
\end{summarybox}

\section*{Formal Restatement}
\addcontentsline{toc}{section}{Formal Restatement}

\begin{formalrestatement}
\textbf{Enriched Yoneda.} $\int_c \mathcal{V}(\mathcal{C}(a,c), F(c)) \cong F(a)$,
$\mathcal{V}$-naturally in $a$.

\textbf{$\mathcal{V}$-adjunction.} $F \dashv_\mathcal{V} G$ iff
$\mathcal{D}(FA,B) \cong \mathcal{C}(A,GB)$ in $\mathcal{V}$.

\textbf{Weighted limit.} $\mathcal{C}(c, \{W,F\}) \cong
[\mathcal{J}^{\mathrm{op}},\mathcal{V}]_\mathcal{V}(W, \mathcal{C}(c,F-))$.

\textbf{Self-enrichment.} Closed symmetric monoidal $(\mathcal{V},\otimes,I,[-,-])$
is a $\mathcal{V}$-category with $\mathcal{V}(A,B) = [A,B]$.

\noindent\textit{Chapter~21 returns to monads via Beck's monadicity theorem.}
\end{formalrestatement}
